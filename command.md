# 1
# Версия
git --version
# Выведите историю проекта через команду 
git log
# Добавление изменение имени
# Переключитесь на нужный коммит
git checkout хеш коммита
git config --global user.name "Emma Paris"
# Добавление изменение адреса
git config --global user.email "eparis@atlassian.com"
# Проверка настроек
git config --global --list
# Настройка формата окончания строк
git config --global core.autocrlf true
git config --global core.safecrlf warn
git config --global core.quotepath off
# Задайте общепринятое название основной ветки в репозитории в Git — main
git config --global init.defaultBranch main
# Git даёт подсказки со всеми возможными командами
git help -g
git help команда
# 2 
# Чтобы проверить текущее состояние репозитория
git status
# В коммит попадут все файлы из текущей папки
git add -A
# Чтобы добавить файл в отслеживаемые
git add README.md  (где git add — это команда, а README.md — имя файла)
# 
git commit -m "add README"  В кавычках принято писать о том,
что изменилось в проекте с последнего снимка.
# Чтобы создать локальный репозиторий
 git init   Проверьте, что появилась папка .git
 # Поместите файл в последний коммит (исправление перезапись)
 git commit --amend -m "error"
# проверить историю проекта
git log --oneline
# покажет всю историю проекта
git log --all --oneline
# Перемещение по истории проекта
git checkout -
# Проверьте, что вернулись к последней версии проекта через команду 
git log –-oneline:

# Чтобы сделать снимок проекта (сформировать коммит)
Добавьте файл или файлы в отслеживаемые командой
git add имя файла
Создайте коммит и подпишите его командой
git commit -m "Подпись коммита"
Чтобы проверить состояние файла
На любом из этапов вы можете выполнить команду
git status
# Внесение изменений в файл Git
Добавьте файл в индекс и подпишите коммит, выполнив последовательно 2 команды
git add README.md и git commit -m "change README"
5 Вновь проверьте статус файла через команду git status. Git выдаст сообщение,
что ничего в проекте не изменилось с последнего сохранения
# Просмотр истории проекта
git log --oneline
# Чтобы посмотреть всю информацию о коммите
 git log без флFага Q — режим вывода закроется
# Исправление коммита
 git commit --amend -m "error" ("error" испрвленный коммит)

# Связь локального и удалённого репозиториев
## Проверьте, что к папке test ещё не привязан никакой удалённый репозиторий
git remote -v
## Связь локального и удалённого репозиториев
git remote add origin ссылка на папку удалённого репозитория
## Если нам когда-нибудь потребуется удалить связь
git remote remove origin 
# Отправка проекта в удалённый репозиторий
git push -u origin main
